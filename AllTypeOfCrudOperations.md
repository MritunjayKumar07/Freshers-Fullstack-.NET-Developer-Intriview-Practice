        ```c#
				.Include(e => e.Addresses) //This method is part of Entity Framework Core and is used for eager loading.
				.Select(e => new EmployeeDetails
				{
					Eid = e.Eid,
					Name = e.Name,
					Position = e.Position,
					Salary = e.Salary ?? 0,
					Addresses = e.Addresses.Select(a => new EmployeeDetails.AddressDetails
					{
						Aid = a.Aid,
						Street = a.Street,
						City = a.City,
						State = a.State,
						PinCode = a.PinCode
					}).ToList()
				}).ToListAsync();

			return Ok(employees);
		}
        ```

        - `Include(e => e.Addresses)`: This tells EF Core to eagerly load the Addresses related to the Employee entities.
        - **Eager loading** means loading the related data (in this case, the Addresses associated with each Employee) along with the primary entity (Employee) in a single query.
